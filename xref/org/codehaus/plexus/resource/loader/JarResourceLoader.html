28  import org.codehaus.plexus.resource.PlexusResource;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  import java.util.LinkedHashMap;
33  import java.util.Map;
34  
35  import javax.inject.Named;
36  
37  /**
38   * @author Jason van Zyl
39   */
40  @Named( JarResourceLoader.ID )
41  public class JarResourceLoader
42          extends AbstractResourceLoader
43  {
44      private static final Logger LOGGER = LoggerFactory.getLogger( JarResourceLoader.class );
45  
46      public static final String ID = "jar";
47  
48      /**
49       * Maps entries to the parent JAR File (key = the entry *excluding* plain directories, value = the JAR URL).
50       */
51      private final Map<String, String> entryDirectory = new LinkedHashMap<>( 559 );
52  
53      /**
54       * Maps JAR URLs to the actual JAR (key = the JAR URL, value = the JAR).
55       */
56      private final Map<String, JarHolder> jarFiles = new LinkedHashMap<>( 89 );
57  
58      private void loadJar( String path )
59      {
60          LOGGER.debug( "JarResourceLoader : trying to load '{}'", path );
61  
62          // Check path information
63          if ( path == null )
64          {
65              LOGGER.error( "JarResourceLoader : can not load JAR - JAR path is null" );
66              return;
67          }
68          if ( !path.startsWith( "jar:" ) )
69          {
70              LOGGER.error( "JarResourceLoader : JAR path must start with jar: -> "
71                      + "see java.net.JarURLConnection for information" );
72              return;
73          }
74          if ( !path.endsWith( "!/" ) )
75          {
76              path += "!/";
77          }
78  
79          // Close the jar if it's already open this is useful for a reload
80          closeJar( path );
81  
82          // Create a new JarHolder
83          JarHolder temp = new JarHolder( path );
84  
85          // Add it's entries to the entryCollection
86          addEntries( temp.getEntries() );
87  
88          // Add it to the Jar table
89          jarFiles.put( temp.getUrlPath(), temp );
90      }
91  
92      /**
93       * Closes a Jar file and set its URLConnection to null.
94       */
95      private void closeJar( String path )
96      {
97          if ( jarFiles.containsKey( path ) )
98          {
99              JarHolder theJar = jarFiles.get( path );
100 
101             theJar.close();
102         }
103     }
104 
105     /**
106      * Copy all the entries into the entryDirectory. It will overwrite any duplicate keys.
107      */
108     private void addEntries( Map<String, String> entries )
109     {
110         entryDirectory.putAll( entries );
111     }
112 
113     /**
114      * Get an {@link PlexusResource} by name.
115      *
116      * @param source name of resource to get
117      * @return PlexusResource containing the resource
118      * @throws ResourceNotFoundException if resource not found.
119      */
120     @Override
121     public PlexusResource getResource( String source )
122             throws ResourceNotFoundException
123     {
124         if ( source == null || source.length() == 0 )
125         {
126             throw new ResourceNotFoundException( "Need to have a resource!" );
127         }
128 
129         /*
130          * if a / leads off, then just nip that :)
131          */
132         if ( source.startsWith( "/" ) )
133         {
134             source = source.substring( 1 );
135         }
136 
137         if ( entryDirectory.containsKey( source ) )
138         {
139             String jarurl = entryDirectory.get( source );
140 
141             final JarHolder holder = jarFiles.get( jarurl );
142             if ( holder != null )
143             {
144                 return holder.getPlexusResource( source );
145             }
146         }
147 
148         throw new ResourceNotFoundException( "JarResourceLoader Error: cannot find resource " + source );
149     }
150 
151     @Override
152     public void addSearchPath( String path )
153     {
154         if ( !paths.contains( path ) )
155         {
156             loadJar( path );
157             paths.add( path );
158         }
159     }
160 }
