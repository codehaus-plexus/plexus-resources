27  import static org.junit.jupiter.api.Assertions.assertEquals;
28  import static org.junit.jupiter.api.Assertions.assertNotNull;
29  import static org.junit.jupiter.api.Assertions.fail;
30  
31  import org.codehaus.plexus.util.IOUtil;
32  
33  import java.io.InputStream;
34  import javax.inject.Inject;
35  
36  /**
37   * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
38   * @version $Id$
39   */
40  public abstract class AbstractResourceLoaderTest
41  {
42      @Inject
43      protected ResourceLoader resourceLoader;
44  
45      protected void assertResource( String name, String expectedContent )
46              throws Exception
47      {
48  
49          InputStream is = resourceLoader.getResource( name ).getInputStream();
50  
51          assertNotNull( is, "The returned input stream is null, name: '" + name + "'." );
52  
53          String actualContent = IOUtil.toString( is, "UTF-8" );
54  
55          assertEquals( expectedContent, actualContent );
56      }
57  
58      protected void assertMissingResource( String name )
59              throws Exception
60      {
61          try
62          {
63              InputStream is = resourceLoader.getResource( name ).getInputStream();
64  
65              String content = IOUtil.toString( is, "UTF-8" );
66  
67              fail( "Expected ResourceNotFoundException while looking for a resource named '" + name + "'. Content:\n"
68                      + content );
69          }
70          catch ( ResourceNotFoundException e )
71          {
72              // expected
73          }
74      }
75  }
